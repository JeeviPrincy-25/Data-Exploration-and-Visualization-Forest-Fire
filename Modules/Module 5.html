import numpy as np
import matplotlib.pyplot as plt
import cartopy.crs as ccrs
import cartopy.feature as cfeature

# Assuming the dataset has columns 'X' and 'Y' for fire locations
lons = df['X'].values  # Longitude values
lats = df['Y'].values  # Latitude values

# Create a figure and axis with the specific projection
fig, ax = plt.subplots(figsize=(10, 7))

# Set up the map with a projection (e.g., PlateCarree for cylindrical projection)
ax = plt.axes(projection=ccrs.PlateCarree())  # Set the projection for the map

# Set the extent (limits of the map)
ax.set_extent([-180, 180, -90, 90], crs=ccrs.PlateCarree())  # Specify extent for longitude/latitude

# Add map features
ax.add_feature(cfeature.LAND, edgecolor='black')  # Land feature
ax.add_feature(cfeature.COASTLINE)  # Coastlines
ax.add_feature(cfeature.BORDERS, linestyle=':')  # Political boundaries

# Plot fire locations on the map
ax.scatter(lons, lats, color='red', s=10, label='Fire Locations', transform=ccrs.PlateCarree())

# Add legend
plt.legend()

# Show the map
plt.show()
